## 快速排序

**思路回顾（和生活类比）：**

想象有一叠扑克牌要按大小排序：

1. 随便选一张牌作为"基准"（pivot）
2. 把比基准小的放左边，比基准大的放右边
3. 对左右两边重复这个过程

**C++ 实现：**

```cpp
#include <iostream>
#include <vector>
using namespace std;

// 划分函数：把数组分成两部分
int partition(vector<int>& arr, int left, int right) {
    int pivot = arr[right];          // 选最右边的元素作为基准
    int i = left - 1;               // i 是"小于基准"的边界

    for (int j = left; j < right; j++) {
        if (arr[j] <= pivot) {
            i++;
            swap(arr[i], arr[j]);   // 把小于基准的移到左边
        }
    }
    swap(arr[i + 1], arr[right]);    // 把基准放到中间的正确位置
    return i + 1;                    // 返回基准的最终位置
}

// 快速排序
void quickSort(vector<int>& arr, int left, int right) {
    if (left < right) {
        int pivotIndex = partition(arr, left, right);  // 划分
        quickSort(arr, left, pivotIndex - 1);           // 排序左半部分
        quickSort(arr, pivotIndex + 1, right);         // 排序右半部分
    }
}

int main() {
    vector<int> arr = {64, 34, 25, 12, 22, 11, 90};

    cout << "排序前: ";
    for (int x : arr) cout << x << " ";
    cout << endl;

    quickSort(arr, 0, arr.size() - 1);

    cout << "排序后: ";
    for (int x : arr) cout << x << " ";
    cout << endl;

    return 0;
}
```

**输出：**
```
排序前: 64 34 25 12 22 11 90
排序后: 11 12 22 25 34 64 90
```

**Lua 版本（对比）：**
```lua
-- Lua 的快速排序（简洁得多！）
function quickSort(arr)
    if #arr <= 1 then return arr end
    local pivot = arr[#arr]
    local left = {}
    local right = {}
    for i = 1, #arr - 1 do
        if arr[i] <= pivot then table.insert(left, arr[i])
        else table.insert(right, arr[i]) end
    end
    local result = {}
    for _, v in ipairs(quickSort(left)) do table.insert(result, v) end
    table.insert(result, pivot)
    for _, v in ipairs(quickSort(right)) do table.insert(result, v) end
    return result
end
```

**对比思考：**
- Lua 用 table，省去了内存管理的麻烦
- C++ 要手动处理数组边界、指针运算
- C++ 更快，Lua 更简洁

## 二分搜索

**思路：** 在有序数组中，每次排除一半

**C++ 实现（非递归）：**
```cpp
#include <iostream>
#include <vector>
using namespace std;

// 二分搜索：找到目标返回下标，没找到返回-1
int binarySearch(vector<int>& arr, int target) {
    int left = 0;
    int right = arr.size() - 1;

    while (left <= right) {
        int mid = left + (right - left) / 2;  // 防止 left+right 溢出

        if (arr[mid] == target) {
            return mid;       // 找到了
        } else if (arr[mid] < target) {
            left = mid + 1;   // 目标在右半部分
        } else {
            right = mid - 1;  // 目标在左半部分
        }
    }

    return -1;  // 没找到
}

int main() {
    vector<int> arr = {1, 3, 5, 7, 9, 11, 13, 15, 17};

    cout << binarySearch(arr, 9) << endl;   // 输出：4（下标从0开始）
    cout << binarySearch(arr, 6) << endl;   // 输出：-1（没找到）

    return 0;
}
```

**和 Lua 的对比：**
```lua
-- Lua 二分搜索
function binarySearch(arr, target)
    local left, right = 1, #arr
    while left <= right do
        local mid = math.floor((left + right) / 2)
        if arr[mid] == target then return mid
        elseif arr[mid] < target then left = mid + 1
        else right = mid - 1 end
    end
    return -1
end
```

## 单向链表

**为什么需要链表？**

数组的大小是固定的，往中间插入元素很麻烦（要移动后面的所有元素）。链表可以动态地增删节点。

```cpp
#include <iostream>
using namespace std;

// 定义链表节点
struct ListNode {
    int data;          // 数据
    ListNode* next;    // 指向下一个节点

    ListNode(int val) : data(val), next(nullptr) {}
};

class LinkedList {
private:
    ListNode* head;    // 头节点（不存数据，只做哨兵）

public:
    LinkedList() {
        head = new ListNode(0);  // 创建头节点（哑节点）
    }

    // 在链表末尾添加
    void pushBack(int val) {
        ListNode* newNode = new ListNode(val);
        ListNode* cur = head;
        while (cur->next != nullptr) {
            cur = cur->next;
        }
        cur->next = newNode;
    }

    // 在链表头部添加
    void pushFront(int val) {
        ListNode* newNode = new ListNode(val);
        newNode->next = head->next;
        head->next = newNode;
    }

    // 删除第一个出现的值
    void remove(int val) {
        ListNode* cur = head;
        while (cur->next != nullptr) {
            if (cur->next->data == val) {
                ListNode* toDelete = cur->next;
                cur->next = toDelete->next;
                delete toDelete;
                return;
            }
            cur = cur->next;
        }
    }

    // 遍历输出
    void print() {
        ListNode* cur = head->next;
        while (cur != nullptr) {
            cout << cur->data << " ";
            cur = cur->next;
        }
        cout << endl;
    }

    // 析构函数：释放所有节点
    ~LinkedList() {
        ListNode* cur = head;
        while (cur != nullptr) {
            ListNode* next = cur->next;
            delete cur;
            cur = next;
        }
    }
};

int main() {
    LinkedList list;
    list.pushBack(10);
    list.pushBack(20);
    list.pushBack(30);
    list.pushFront(5);

    list.print();    // 输出：5 10 20 30

    list.remove(20);
    list.print();    // 输出：5 10 30

    return 0;
}
```

**图解链表结构：**

```
head → [5] → [10] → [30] → nullptr
```

**和 Lua 的 table 实现链表的对比：**

```lua
-- Lua 用 table 实现链表
local node1 = {value = 10, next = nil}
local node2 = {value = 20, next = nil}
node1.next = node2
-- Lua 不需要手动释放内存（GC自动处理）
```

## 练习：用 C++ 实现栈

**栈的特性：LIFO（后进先出）**

```cpp
#include <iostream>
#include <vector>
using namespace std;

class Stack {
private:
    vector<int> data;  // 用 vector 存储

public:
    // 入栈
    void push(int val) {
        data.push_back(val);
    }

    // 出栈
    int pop() {
        if (data.empty()) {
            cout << "栈为空！" << endl;
            return -1;
        }
        int top = data.back();
        data.pop_back();
        return top;
    }

    // 查看栈顶
    int top() {
        if (data.empty()) return -1;
        return data.back();
    }

    // 是否为空
    bool empty() {
        return data.empty();
    }

    // 大小
    int size() {
        return data.size();
    }
};

int main() {
    Stack s;
    s.push(10);
    s.push(20);
    s.push(30);

    cout << "栈顶: " << s.top() << endl;  // 30
    cout << "出栈: " << s.pop() << endl;   // 30
    cout << "栈顶: " << s.top() << endl;   // 20

    return 0;
}
```

**用栈实现浏览器前进/后退：**
```cpp
Stack backStack;    // 后退栈
Stack forwardStack; // 前进栈

void go(string url) {
    backStack.push(currentUrl);  // 当前页面入后退栈
    currentUrl = url;
    while (!forwardStack.empty()) {  // 清空前进栈
        forwardStack.pop();
    }
}

void back() {
    if (backStack.empty()) return;
    forwardStack.push(currentUrl);
    currentUrl = backStack.pop();
}
```
