## 项目目标

做一个完整的回合制战斗游戏，涵盖前面学的大部分内容：

```
===== 回合制战斗 =====
玩家: 战士  HP:100/100  攻击:25
敌人: 哥布林 HP:80/80    攻击:15

1. 普通攻击
2. 技能: 盾击（消耗10MP）
3. 使用药水（剩余3瓶）
4. 查看敌人状态

选择行动: 1
战士 攻击 哥布林，造成 25 点伤害！
哥布林 HP: 80 → 55

===== 敌人回合 =====
哥布林 攻击 战士，造成 15 点伤害！
战士 HP: 100 → 85
```

**涉及的技术点：**
- class（角色、技能、药水）
- 继承（战士、法师有各自属性）
- 多态（不同职业攻击方式不同）
- vector / map（存储技能、背包）
- 文件读写（存档/读档）
- 循环和条件判断

## 第一步：定义角色基类

```cpp
class Character {
protected:
    string name;
    int hp;
    int maxHp;
    int attackPower;

public:
    Character(string n, int hpInit, int ap)
        : name(n), hp(hpInit), maxHp(hpInit), attackPower(ap) {}

    // 虚函数：子类必须实现
    virtual void attackTarget(Character* target) = 0;
    virtual string getJobName() = 0;

    // 普通方法
    void takeDamage(int dmg) {
        hp -= dmg;
        if (hp < 0) hp = 0;
    }

    bool isAlive() {
        return hp > 0;
    }

    void showStatus() {
        cout << name << " (" << getJobName() << ") "
             << "HP:" << hp << "/" << maxHp
             << " 攻击:" << attackPower << endl;
    }

    string getName() { return name; }
    int getHp() { return hp; }
    int getAttackPower() { return attackPower; }

    virtual ~Character() {}
};
```

## 第二步：定义战士和法师

```cpp
class Warrior : public Character {
private:
    int armor;  // 战士特有的护甲

public:
    Warrior(string n)
        : Character(n, 100, 25), armor(10) {}

    void attackTarget(Character* target) override {
        int dmg = attackPower;
        target->takeDamage(dmg);
        cout << name << " 挥剑攻击 " << target->getName()
             << "，造成 " << dmg << " 点伤害！" << endl;
    }

    string getJobName() override {
        return "战士";
    }
};

class Mage : public Character {
private:
    int mp;
    int maxMp;

public:
    Mage(string n)
        : Character(n, 60, 35), mp(50), maxMp(50) {}

    void attackTarget(Character* target) override {
        int dmg = attackPower;
        target->takeDamage(dmg);
        cout << name << " 释放火球术 " << target->getName()
             << "，造成 " << dmg << " 点伤害！" << endl;
    }

    void castSkill(Character* target) {
        if (mp >= 20) {
            int dmg = attackPower * 2;
            target->takeDamage(dmg);
            mp -= 20;
            cout << name << " 施放奥术弹幕 " << target->getName()
                 << "，造成 " << dmg << " 点伤害！剩余MP:" << mp << endl;
        } else {
            cout << "MP不足，无法施法！" << endl;
        }
    }

    string getJobName() override {
        return "法师";
    }
};
```

## 第三步：定义技能系统

```cpp
struct Skill {
    string name;
    int damage;       // 伤害值
    int mpCost;       // MP消耗
    int cooldown;     // 冷却回合
    int currentCd;   // 当前冷却

    Skill(string n, int dmg, int mp, int cd)
        : name(n), damage(dmg), mpCost(mp), cooldown(cd), currentCd(0) {}

    bool isReady() {
        return currentCd == 0;
    }

    void use() {
        currentCd = cooldown;
    }

    void decreaseCd() {
        if (currentCd > 0) currentCd--;
    }
};

class SkillSystem {
private:
    vector<Skill> skills;

public:
    void addSkill(string name, int dmg, int mp, int cd) {
        skills.emplace_back(name, dmg, mp, cd);
    }

    void showSkills() {
        cout << "===== 技能列表 =====" << endl;
        for (int i = 0; i < skills.size(); i++) {
            Skill& s = skills[i];
            string status = s.isReady() ? "[可使用]" : "[冷却中:" + to_string(s.currentCd) + "回合]";
            cout << i+1 << ". " << s.name << " 伤害:" << s.damage
                 << " MP:" << s.mpCost << " " << status << endl;
        }
    }

    Skill* getSkill(int index) {
        if (index >= 0 && index < (int)skills.size()) {
            return &skills[index];
        }
        return nullptr;
    }

    void decreaseAllCd() {
        for (auto& s : skills) {
            s.decreaseCd();
        }
    }
};
```

## 第四步：定义药水系统

```cpp
class Potion {
private:
    vector<int> potions;  // 药水数量列表

public:
    Potion() {
        potions.push_back(3);   // 生命药水 x3
        potions.push_back(2);   // 魔法药水 x2
    }

    void usePotion(int type) {
        if (type == 1 && potions[0] > 0) {
            potions[0]--;
            cout << "使用生命药水，恢复 30 HP" << endl;
        } else if (type == 2 && potions[1] > 0) {
            potions[1]--;
            cout << "使用魔法药水，恢复 20 MP" << endl;
        } else {
            cout << "没有药水了！" << endl;
        }
    }

    void showPotions() {
        cout << "===== 背包 =====" << endl;
        cout << "1. 生命药水（剩余 " << potions[0] << " 瓶）" << endl;
        cout << "2. 魔法药水（剩余 " << potions[1] << " 瓶）" << endl;
    }

    int getPotionCount(int type) {
        if (type >= 0 && type < (int)potions.size()) {
            return potions[type];
        }
        return 0;
    }
};
```

## 第五步：主战斗循环

```cpp
#include <iostream>
#include <vector>
#include <string>
using namespace std;

int main() {
    cout << "===== 选择你的职业 =====" << endl;
    cout << "1. 战士（高血量，近战）" << endl;
    cout << "2. 法师（高攻击，低血量）" << endl;

    int choice;
    cin >> choice;

    Character* player;
    if (choice == 1) {
        player = new Warrior("玩家");
    } else {
        player = new Mage("玩家");
    }

    Character* enemy = new Warrior("哥布林");

    cout << "\n===== 战斗开始！ =====" << endl;
    player->showStatus();
    enemy->showStatus();
    cout << endl;

    int round = 1;

    while (player->isAlive() && enemy->isAlive()) {
        cout << "===== 第 " << round << " 回合 =====" << endl;
        player->showStatus();
        enemy->showStatus();
        cout << endl;

        // 玩家行动
        cout << "1. 普通攻击" << endl;
        cout << "2. 查看敌人状态" << endl;
        cout << "3. 查看背包" << endl;

        int action;
        cin >> action;

        if (action == 1) {
            player->attackTarget(enemy);
        } else if (action == 2) {
            enemy->showStatus();
            round--;
        } else if (action == 3) {
            cout << "背包功能" << endl;
            round--;
        }

        // 敌人反击
        if (enemy->isAlive()) {
            cout << "\n===== 敌人回合 =====" << endl;
            enemy->attackTarget(player);
        }

        cout << endl;
        round++;
    }

    if (player->isAlive()) {
        cout << "===== 胜利！ =====" << endl;
        cout << "你击败了 " << enemy->getName() << "！" << endl;
    } else {
        cout << "===== 失败 =====" << endl;
        cout << "你被 " << enemy->getName() << " 击败了..." << endl;
    }

    delete player;
    delete enemy;

    return 0;
}
```

## 扩展1：存档和读档

```cpp
// 存档
void saveGame(Character* player, Character* enemy, int round) {
    ofstream out("savegame.txt");
    out << player->getName() << " " << player->getHp() << " " << round << endl;
    out << enemy->getName() << " " << enemy->getHp() << endl;
    out.close();
    cout << "游戏已保存" << endl;
}

// 读档
void loadGame(Character*& player, Character*& enemy, int& round) {
    ifstream in("savegame.txt");
    if (!in) {
        cout << "没有存档" << endl;
        return;
    }

    string name;
    int hp, roundLoad;
    in >> name >> hp >> roundLoad;

    if (player) delete player;
    if (enemy) delete enemy;

    player = new Warrior(name);
    player->takeDamage(100 - hp);  // 扣血到当前HP
    enemy = new Warrior("哥布林");
    enemy->takeDamage(100 - 50);  // 假设敌人之前有50HP
    round = roundLoad;

    in.close();
    cout << "游戏已加载，第 " << round << " 回合" << endl;
}
```
